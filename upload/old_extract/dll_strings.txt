        // --- Validate feed URL (server-side via API) ---
        function validateFeed() {
            var url = document.getElementById('podcastFeedUrl').value.trim();
            if (!url) {
                showValidationResult(false, 'Ingresa una URL de feed RSS.');
                return;
            }
            // Basic URL validation
            try {
                var uri = new URL(url);
                if (uri.protocol !== 'http:' && uri.protocol !== 'https:') {
                    showValidationResult(false, 'La URL debe comenzar con http:// o https://.');
                    return;
                }
            } catch (e) {
                showValidationResult(false, 'La URL proporcionada no es v
lida.');
                return;
            }
            hideValidationResult();
            var validateBtn = document.getElementById('btnValidateFeed');
            validateBtn.disabled = true;
            validateBtn.textContent = 'Validando...';
            // Validate via server-side API (avoids CORS issues)
            ApiClient.ajax({
                type: 'POST',
                url: ApiClient.getUrl('Plugins/Podcasts/ValidateFeed') + '?url=' + encodeURIComponent(url),
                dataType: 'json'
            }).then(function (result) {
                validateBtn.disabled = false;
                validateBtn.innerHTML = '<span class="material-icons button-icon button-icon-left" style="font-size:inherit;">check_circle</span> Validar Feed';
                if (result.success) {
                    isFormValidated = true;
                    showValidationResult(true, result.message);
                } else {
                    isFormValidated = false;
                    showValidationResult(false, result.error || 'Error desconocido.');
                }
            }).catch(function (error) {
                validateBtn.disabled = false;
                validateBtn.innerHTML = '<span class="material-icons button-icon button-icon-left" style="font-size:inherit;">check_circle</span> Validar Feed';
                isFormValidated = false;
                showValidationResult(false, 'No se pudo conectar al servidor para validar el feed.');
            });
        }
        // --- Save feed (add or update) ---
        function saveFeed() {
            var name = document.getElementById('podcastName').value.trim();
            var frequency = parseInt(document.getElementById('podcastFrequency').value, 10);
            var feedUrl = document.getElementById('podcastFeedUrl').value.trim();
            var autoDelete = parseInt(document.getElementById('podcastAutoDelete').value, 10);
            var includeInAutoPlaylist = document.getElementById('podcastAutoPlaylist').checked;
            // Validate fields
            if (!name) {
                Dashboard.alert({ title: 'Campo requerido', message: 'Ingresa el nombre del podcast.' });
                return;
            }
            if (!feedUrl) {
                Dashboard.alert({ title: 'Campo requerido', message: 'Ingresa el enlace del feed RSS.' });
                return;
            }
            // Require validation for new feeds
            if (!editingFeedId && !isFormValidated) {
                Dashboard.alert({ title: 'Validaci
n requerida', message: 'Debes validar el feed RSS antes de agregarlo.' });
                return;
            }
            // Check for duplicate name (excluding current feed when editing)
            var duplicateName = currentConfig.feeds.find(function (f) {
                return f.name.toLowerCase() === name.toLowerCase() && f.id !== editingFeedId;
            });
            if (duplicateName) {
                Dashboard.alert({ title: 'Nombre duplicado', message: 'Ya existe un podcast con ese nombre.' });
                return;
            }
            if (editingFeedId) {
                // Update existing feed
                var feedIndex = currentConfig.feeds.findIndex(function (f) { return f.id === editingFeedId; });
                if (feedIndex >= 0) {
                    currentConfig.feeds[feedIndex].name = name;
                    currentConfig.feeds[feedIndex].frequency = frequency;
                    currentConfig.feeds[feedIndex].feedUrl = feedUrl;
                    currentConfig.feeds[feedIndex].autoDelete = autoDelete;
                    currentConfig.feeds[feedIndex].includeInAutoPlaylist = includeInAutoPlaylist;
                }
            } else {
                // Add new feed
                currentConfig.feeds.push({
                    id: crypto.randomUUID ? crypto.randomUUID() : 'xxxxxxxx-xxxx-4xxx-yxxx-xxxxxxxxxxxx'.replace(/[xy]/g, function (c) { var r = Math.random() * 16 | 0; return (c === 'x' ? r : (r & 0x3 | 0x8)).toString(16); }),
                    name: name,
                    frequency: frequency,
                    feedUrl: feedUrl,
                    autoDelete: autoDelete,
                    includeInAutoPlaylist: includeInAutoPlaylist,
                    lastUpdateDate: null
                });
            }
            saveConfig();
            hideForm();
            renderFeedsList();
            updateStatusBanner();
        }
